function addOrder()
{

            	alert(error);

                errors['add_order_connection_error'] = error;
            }
    });

    $('label.error').text('');
    $('label.error').addClass('hide');
    $('#error-box').addClass('hide');

    $.each( errors, function (key, val)
    {

}
